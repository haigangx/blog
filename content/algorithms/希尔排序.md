# 希尔排序

```
void shell(vector<int>& nums)
{
    int size = nums.size();
    int h = 1;

    while (h < (size / 3))
    {
        h = 3 * h + 1;
    }

    while (h >= 1)
    {
        for (int i = h; i < size; i++)
        {
            for (int j = i; j >= h && nums[j] < nums[j - h]; j -= h)
            {
                swap(nums, j, j - h);
            }
            for (int k = 0; k < nums.size(); k++)
                cout << nums[k] << " ";
            cout << endl;
        }
        h /= 3;
    }
}
```
